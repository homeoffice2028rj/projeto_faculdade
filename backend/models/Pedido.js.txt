const mongoose =
    require("mongoose");


const pedidoSchema =
    new mongoose.Schema({

        // CLIENTE

        cliente: {

            type: String,

            required: true

        },


        telefone: {

            type: String,

            required: true

        },


        // ENDEREÇO

        endereco: {

            rua: {

                type: String,

                required: true

            },

            numero: {

                type: String,

                required: true

            },

            bairro: {

                type: String,

                required: true

            },

            cidade: {

                type: String,

                required: true

            },

            cep: {

                type: String,

                required: true

            },

            complemento: {

                type: String,

                default: ""

            }

        },


        // PRODUTOS

        produtos: [

            {

                nome: String,

                preco: Number,

                quantidade: {

                    type: Number,

                    default: 1

                }

            }

        ],


        // TOTAL PEDIDO

        total: {

            type: Number,

            required: true

        },


        // OBSERVAÇÃO

        observacao: {

            type: String,

            default: ""

        },


        // STATUS PEDIDO

        status: {

            type: String,

            enum: [

                "pendente",

                "preparando",

                "saiu_entrega",

                "entregue",

                "cancelado"

            ],

            default: "pendente"

        },


        // DATA PEDIDO

        criadoEm: {

            type: Date,

            default: Date.now

        }

    });


module.exports =
    mongoose.model(
        "Pedido",
        pedidoSchema
    );
