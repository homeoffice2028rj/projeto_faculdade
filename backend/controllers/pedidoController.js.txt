const atualizarStatus =
    async (req, res) => {

        try {

            const pedido =
                await Pedido.findByIdAndUpdate(

                    req.params.id,

                    {

                        status:
                            req.body.status

                    },

                    {

                        new: true

                    }

                );


            res.status(200).json({

                mensagem:
                    "Status atualizado",

                pedido

            });

        } catch (error) {

            res.status(500).json({

                mensagem:
                    "Erro atualizar status"

            });

        }

};
